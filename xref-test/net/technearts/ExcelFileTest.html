22    @Test
23    public void testExcelFile() {
24  
25      try (ExcelFile xls =
26          new ExcelFile(new File(this.getClass().getResource("/Pasta1.xlsx").toURI()))) {
27        xls.write("Plan1", 0, 0, "A");
28        xls.write("Plan1", 0, 1, "B");
29        xls.write("Plan1", 0, 2, "C");
30        xls.write("Plan1", 1, 0, 1);
31        xls.write("Plan1", 1, 1, 2);
32        xls.write("Plan1", 1, 2, 3);
33        assertEquals(xls.read("Plan1", 0, 0, ExcelFileTest::stringTest), "A");
34        assertEquals(xls.read("Plan1", 0, 1, ExcelFileTest::stringTest), "B");
35        assertEquals(xls.read("Plan1", 0, 2, ExcelFileTest::stringTest), "C");
36        assertEquals(xls.read("Plan1", 1, 0, Double::valueOf, ExcelFileTest::doubleTest), 1.0d,
37            0.001d);
38        assertEquals(xls.read("Plan1", 1, 1, Double::valueOf, ExcelFileTest::doubleTest), 2.0d,
39            0.001d);
40        assertEquals(xls.read("Plan1", 1, 2, Double::valueOf, ExcelFileTest::doubleTest), 3.0d,
41            0.001d);
42      } catch (URISyntaxException | IOException e) {
43        fail(e.getMessage());
44      }
45      assertTrue(true);
46    }

48    @Test
49    public void testExcelSheet() {
50      URL url = this.getClass().getResource("/Pasta1.xlsx");
51      try {
52        URI uri = url.toURI();
53        try (ExcelFile xls = new ExcelFile(new File(uri))) {
54          ExcelFile.ExcelSheet sheet = xls.sheet("Plan2");
55          sheet.write(0, 0, "A");
56          sheet.write(0, 1, "B");
57          sheet.write(0, 2, "C");
58          sheet.write(1, 0, 1);
59          sheet.write(1, 1, 2);
60          sheet.write(1, 2, 3);
61          assertEquals(sheet.read(0, 0, ExcelFileTest::stringTest), "A");
62          assertEquals(sheet.read(0, 1, ExcelFileTest::stringTest), "B");
63          assertEquals(sheet.read(0, 2, ExcelFileTest::stringTest), "C");
64          assertEquals(sheet.read(1, 0, Double::valueOf, ExcelFileTest::doubleTest), 1.0d, 0.001d);
65          assertEquals(sheet.read(1, 1, Double::valueOf, ExcelFileTest::doubleTest), 2.0d, 0.001d);
66          assertEquals(sheet.read(1, 2, Double::valueOf, ExcelFileTest::doubleTest), 3.0d, 0.001d);
67        } catch (IOException e) {
68          fail(e.getMessage());
69        }
70      } catch (URISyntaxException e1) {
71        // TODO Auto-generated catch block
72        e1.printStackTrace();
73      }
74      assertTrue(true);
75    }
